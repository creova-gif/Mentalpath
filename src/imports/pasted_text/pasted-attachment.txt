// ── CLIENT FACTORY ────────────────────────────────────────────

let _supabase: SupabaseClient | null = null

export function getSupabase(): SupabaseClient {
  if (!_supabase) {
    const url = process.env.NEXT_PUBLIC_SUPABASE_URL
    const key = process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY
    if (!url || !key) throw new Error('Missing Supabase env vars')
    _supabase = createClient(url, key, {
      auth: {
        persistSession: true,
        autoRefreshToken: true,
      },
    })
  }
  return _supabase
}

// ── AUTH ──────────────────────────────────────────────────────

export const auth = {
  async signUp(email: string, password: string, fullName: string) {
    const sb = getSupabase()
    const { data, error } = await sb.auth.signUp({
      email,
      password,
      options: {
        data: { full_name: fullName },
        emailRedirectTo: `${process.env.NEXT_PUBLIC_APP_URL}/dashboard`,
      },
    })
    if (error) throw error
    return data
  },

  async signIn(email: string, password: string) {
    const sb = getSupabase()
    const { data, error } = await sb.auth.signInWithPassword({ email, password })
    if (error) throw error
    return data
  },

  async signOut() {
    const sb = getSupabase()
    const { error } = await sb.auth.signOut()
    if (error) throw error
  },

  async getUser(): Promise<User | null> {
    const sb = getSupabase()
    const { data: { user } } = await sb.auth.getUser()
    return user
  },

  async resetPassword(email: string) {
    const sb = getSupabase()
    const { error } = await sb.auth.resetPasswordForEmail(email, {
      redirectTo: `${process.env.NEXT_PUBLIC_APP_URL}/reset-password`,
    })
    if (error) throw error
  },
}

// ── THERAPIST API ─────────────────────────────────────────────

export const therapistAPI = {
  async get(): Promise<Therapist> {
    const sb = getSupabase()
    const user = await auth.getUser()
    if (!user) throw new Error('Not authenticated')
    const { data, error } = await sb
      .from('therapists')
      .select('*')
      .eq('id', user.id)
      .single()
    if (error) throw error
    return data
  },

  async create(profile: Partial<Therapist>): Promise<Therapist> {
    const sb = getSupabase()
    const user = await auth.getUser()
    if (!user) throw new Error('Not authenticated')
    const { data, error } = await sb
      .from('therapists')
      .insert({ id: user.id, email: user.email, ...profile })
      .select()
      .single()
    if (error) throw error
    return data
  },

  async update(updates: Partial<Therapist>): Promise<Therapist> {
    const sb = getSupabase()
    const user = await auth.getUser()
    if (!user) throw new Error('Not authenticated')
    const { data, error } = await sb
      .from('therapists')
      .update(updates)
      .eq('id', user.id)
      .select()
      .single()
    if (error) throw error
    return data
  },

  // Check subscription access
  canUseAI(tier: SubscriptionTier): boolean {
    return tier === 'solo' || tier === 'group'
  },

  canAddClients(tier: SubscriptionTier, currentCount: number): boolean {
    if (tier === 'free') return currentCount < 1
    return true
  },
}

// ── CLIENTS API ───────────────────────────────────────────────

export const clientsAPI = {
  async list(filters?: { status?: ClientStatus }): Promise<Client[]> {
    const sb = getSupabase()
    let query = sb.from('clients').select('*').order('last_name')
    if (filters?.status) query = query.eq('status', filters.status)
    const { data, error } = await query
    if (error) throw error
    return (data || []).map(c => ({ ...c, full_name: `${c.first_name} ${c.last_name}` }))
  },

  async get(clientId: string): Promise<Client> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('clients')
      .select('*')
      .eq('id', clientId)
      .single()
    if (error) throw error
    return { ...data, full_name: `${data.first_name} ${data.last_name}` }
  },

  async create(client: Partial<Client>): Promise<Client> {
    const sb = getSupabase()
    const user = await auth.getUser()
    if (!user) throw new Error('Not authenticated')
    const { data, error } = await sb
      .from('clients')
      .insert({ therapist_id: user.id, ...client })
      .select()
      .single()
    if (error) throw error
    return { ...data, full_name: `${data.first_name} ${data.last_name}` }
  },

  async update(clientId: string, updates: Partial<Client>): Promise<Client> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('clients')
      .update(updates)
      .eq('id', clientId)
      .select()
      .single()
    if (error) throw error
    return { ...data, full_name: `${data.first_name} ${data.last_name}` }
  },

  async incrementSessionCount(clientId: string): Promise<void> {
    const sb = getSupabase()
    const { error } = await sb.rpc('increment_session_count', { client_id: clientId })
    if (error) {
      // Fallback if RPC doesn't exist yet
      const { data: client } = await sb.from('clients').select('session_count').eq('id', clientId).single()
      await sb.from('clients').update({
        session_count: (client?.session_count || 0) + 1,
        last_session_at: new Date().toISOString(),
      }).eq('id', clientId)
    }
  },

  async search(query: string): Promise<Client[]> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('clients')
      .select('*')
      .or(`first_name.ilike.%${query}%,last_name.ilike.%${query}%,email.ilike.%${query}%`)
      .order('last_name')
      .limit(20)
    if (error) throw error
    return (data || []).map(c => ({ ...c, full_name: `${c.first_name} ${c.last_name}` }))
  },
}

// ── NOTES API ─────────────────────────────────────────────────

export const notesAPI = {
  async list(clientId?: string): Promise<SessionNote[]> {
    const sb = getSupabase()
    let query = sb
      .from('session_notes')
      .select('*, client:clients(first_name, last_name)')
      .order('session_date', { ascending: false })
    if (clientId) query = query.eq('client_id', clientId)
    const { data, error } = await query
    if (error) throw error
    return data || []
  },

  async listDue(): Promise<SessionNote[]> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('session_notes')
      .select('*, client:clients(first_name, last_name)')
      .eq('is_draft', true)
      .eq('is_locked', false)
      .order('session_date', { ascending: false })
    if (error) throw error
    return data || []
  },

  async get(noteId: string): Promise<SessionNote> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('session_notes')
      .select('*, client:clients(*)')
      .eq('id', noteId)
      .single()
    if (error) throw error
    return data
  },

  async create(note: Partial<SessionNote>): Promise<SessionNote> {
    const sb = getSupabase()
    const user = await auth.getUser()
    if (!user) throw new Error('Not authenticated')
    const { data, error } = await sb
      .from('session_notes')
      .insert({ therapist_id: user.id, ...note })
      .select()
      .single()
    if (error) throw error
    return data
  },

  async save(noteId: string, updates: Partial<SessionNote>): Promise<SessionNote> {
    const sb = getSupabase()
    // Can't edit locked notes
    const { data: existing } = await sb.from('session_notes').select('is_locked').eq('id', noteId).single()
    if (existing?.is_locked) throw new Error('Note is locked and cannot be edited')
    const { data, error } = await sb
      .from('session_notes')
      .update({ ...updates, is_draft: true })
      .eq('id', noteId)
      .select()
      .single()
    if (error) throw error
    return data
  },

  async lock(noteId: string): Promise<SessionNote> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('session_notes')
      .update({ is_draft: false, is_locked: true, locked_at: new Date().toISOString() })
      .eq('id', noteId)
      .select()
      .single()
    if (error) throw error
    return data
  },

  // Call AI assist edge function
  async getAIDraft(params: {
    sessionId: string
    noteFormat: NoteFormat
    section1: string
    section2: string
    section3: string
    sessionContext?: string
  }): Promise<{ draft: string; format: string; disclaimer: string }> {
    const sb = getSupabase()
    const { data: { session } } = await sb.auth.getSession()
    if (!session) throw new Error('Not authenticated')

    const res = await fetch(
      `${process.env.NEXT_PUBLIC_SUPABASE_URL}/functions/v1/ai-note-assist`,
      {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': `Bearer ${session.access_token}`,
          'apikey': process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
        },
        body: JSON.stringify({
          session_id: params.sessionId,
          note_format: params.noteFormat,
          section_1: params.section1,
          section_2: params.section2,
          section_3: params.section3,
          session_context: params.sessionContext,
        }),
      }
    )

    if (!res.ok) {
      const err = await res.json()
      throw new Error(err.error || 'AI assist failed')
    }

    return res.json()
  },
}

// ── INVOICES API ──────────────────────────────────────────────

export const invoicesAPI = {
  async list(filters?: { status?: InvoiceStatus; clientId?: string }): Promise<Invoice[]> {
    const sb = getSupabase()
    let query = sb
      .from('invoices')
      .select('*, client:clients(first_name, last_name)')
      .order('invoice_date', { ascending: false })
    if (filters?.status) query = query.eq('status', filters.status)
    if (filters?.clientId) query = query.eq('client_id', filters.clientId)
    const { data, error } = await query
    if (error) throw error
    return data || []
  },

  async create(invoice: Partial<Invoice>): Promise<Invoice> {
    const sb = getSupabase()
    const user = await auth.getUser()
    if (!user) throw new Error('Not authenticated')

    // Auto-generate invoice number
    const { data: numData } = await sb.rpc('generate_invoice_number', { p_therapist_id: user.id })
    const invoiceNumber = numData || `INV-${Date.now()}`

    const { data, error } = await sb
      .from('invoices')
      .insert({
        therapist_id: user.id,
        invoice_number: invoiceNumber,
        ...invoice,
      })
      .select('*, client:clients(first_name, last_name)')
      .single()
    if (error) throw error
    return data
  },

  async markPaid(invoiceId: string, paymentIntentId?: string): Promise<Invoice> {
    const sb = getSupabase()
    const { data, error } = await sb
      .from('invoices')
      .update({
        status: 'paid',
        paid_at: new Date().toISOString(),
        stripe_payment_intent_id: paymentIntentId,
      })
      .eq('id', invoiceId)
      .select()
      .single()
    if (error) throw error
    return data
  },

  // Monthly revenue summary
  async getMonthlySummary(year: number, month: number) {
    const sb = getSupabase()
    const start = `${year}-${String(month).padStart(2, '0')}-01`
    const end = new Date(year, month, 0).toISOString().split('T')[0]
    const { data, error } = await sb
      .from('invoices')
      .select('total, status, paid_at')
      .gte('invoice_date', start)
      .lte('invoice_date', end)
    if (error) throw error
    const collected = (data || []).filter(i => i.status === 'paid').reduce((s, i) => s + i.total, 0)
    const outstanding = (data || []).filter(i => i.status === 'pending').reduce((s, i) => s + i.total, 0)
    const overdue = (data || []).filter(i => i.status === 'overdue').reduce((s, i) => s + i.total, 0)
    return { collected, outstanding, overdue, total: collected + outstanding + overdue }
  },
}

// ── APPOINTMENTS API ──────────────────────────────────────────

export const appointmentsAPI = {
  async listToday(): Promise<Appointment[]> {
    const sb = getSupabase()
    const today = new Date().toISOString().split('T')[0]
    const { data, error } = await sb
      .from('appointments')
      .select('*, client:clients(first_name, last_name, pronouns)')

      .single()

    return data
  },
}

// ── DASHBOARD STATS ───────────────────────────────────────────

export async function getDashboardStats() {
  const sb = getSupabase()
  const today = new Date().toISOString().split('T')[0]
  const monthStart = `${today.slice(0, 7)}-01`

  const [
    { count: activeClients },
    { data: todayAppts },
    { data: dueNotes },
    monthlySummary,
  ] = await Promise.all([
    sb.from('clients').select('*', { count: 'exact', head: true }).eq('status', 'active'),
    sb.from('appointments').select('id').gte('scheduled_at', `${today}T00:00:00`).lte('scheduled_at', `${today}T23:59:59`),
    sb.from('session_notes').select('id').eq('is_draft', true).eq('is_locked', false),
    invoicesAPI.getMonthlySummary(new Date().getFullYear(), new Date().getMonth() + 1),
  ])

  return {
    activeClients: activeClients || 0,
    sessionsToday: todayAppts?.length || 0,
    notesDue: dueNotes?.length || 0,
    monthlyCollected: monthlySummary.collected,
    monthlyOutstanding: monthlySummary.outstanding,
  }
}

// ── PORTAL API (no auth — token-based) ───────────────────────

export const portalAPI = {
  // Get client by portal token (called from portal page)
  async getClientByToken(token: string) {
    // Use service role for this — called from server component only
    const url = process.env.NEXT_PUBLIC_SUPABASE_URL!
    const serviceKey = process.env.SUPABASE_SERVICE_ROLE_KEY!
    const sb = createClient(url, serviceKey)

    const { data, error } = await sb
      .from('clients')
      .select('*, therapist:therapists(full_name, credentials, registration_number, practice_name, hourly_rate)')
      .eq('portal_access_token', token)
      .gt('portal_token_expires_at', new Date().toISOString())
      .single()

    if (error || !data) return null
    return data
  },

  // Submit intake form (no auth required)
  async submitIntakeForm(params: {
    token: string
    formData: Record<string, unknown>
    bookingDate?: string
    bookingTime?: string
    sessionFormat?: string
    signatureName?: string
    consents: { cb1: boolean; cb2: boolean; cb3: boolean; cb4: boolean }
  }) {
    const url = process.env.NEXT_PUBLIC_SUPABASE_URL!
    const serviceKey = process.env.SUPABASE_SERVICE_ROLE_KEY!
    const sb = createClient(url, serviceKey)

    // Get client by token
    const client = await portalAPI.getClientByToken(params.token)
    if (!client) throw new Error('Invalid or expired portal link')

    // Save intake form
    const { data, error } = await sb
      .from('intake_forms')
      .insert({
        client_id: client.id,
        therapist_id: client.therapist_id,
        template_type: client.intake_template,
        form_data: params.formData,
        booking_date: params.bookingDate,
        booking_time: params.bookingTime,
        session_format: params.sessionFormat,
        signature_name: params.signatureName,
        consent_cb1: params.consents.cb1,
        consent_cb2: params.consents.cb2,
        consent_cb3: params.consents.cb3,
        consent_cb4: params.consents.cb4,
      })
      .select()
      .single()

    if (error) throw error

    // Mark client consent signed
    await sb.from('clients').update({
      consent_signed_at: new Date().toISOString(),
    }).eq('id', client.id)

    return data
  },
}
