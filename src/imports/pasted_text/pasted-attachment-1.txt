#!/bin/bash
# ============================================================
# MentalPath — Project Bootstrap Script
# Run from your projects folder: bash bootstrap.sh
# ============================================================

echo "🍁 Bootstrapping MentalPath..."
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

# ── 1. Create Next.js project ──────────────────────────────
npx create-next-app@latest mentalpath \
  --typescript \
  --tailwind \
  --eslint \
  --app \
  --src-dir \
  --import-alias "@/*" \
  --no-git

cd mentalpath

# ── 2. Install dependencies ────────────────────────────────
echo "📦 Installing dependencies..."
npm install \
  @supabase/supabase-js \
  @supabase/ssr \
  stripe \
  @stripe/stripe-js \
  @anthropic-ai/sdk \
  resend \
  twilio \
  @react-pdf/renderer \
  date-fns \
  react-hook-form \
  zod \
  @hookform/resolvers \
  lucide-react \
  clsx \
  tailwind-merge

npm install --save-dev \
  @types/node \
  supabase

echo "✅ Dependencies installed"

# ── 3. Copy lib files ─────────────────────────────────────
echo "📁 Setting up lib files..."
mkdir -p src/lib src/types src/app/dashboard src/app/portal src/app/api/webhooks

# Copy the supabase.ts lib file
# (You already have this from the build session)
cp ../mentalpath-src/lib/supabase.ts src/lib/supabase.ts

# ── 4. Set up environment ─────────────────────────────────
echo "🔐 Setting up environment..."
cat > .env.local << 'EOF'
NEXT_PUBLIC_SUPABASE_URL=https://hkhwgbkijepsxtixdmrs.supabase.co
NEXT_PUBLIC_SUPABASE_ANON_KEY=PASTE_YOUR_ANON_KEY_HERE
SUPABASE_SERVICE_ROLE_KEY=PASTE_YOUR_SERVICE_ROLE_KEY_HERE
STRIPE_SECRET_KEY=sk_test_PASTE_YOUR_KEY
NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY=pk_test_PASTE_YOUR_KEY
STRIPE_WEBHOOK_SECRET=whsec_PASTE_AFTER_STRIPE_LISTEN
STRIPE_SOLO_PRICE_ID=price_PASTE_AFTER_CREATING_PRODUCT
STRIPE_GROUP_PRICE_ID=price_PASTE_AFTER_CREATING_PRODUCT
ANTHROPIC_API_KEY=sk-ant-PASTE_YOUR_KEY
RESEND_API_KEY=re_PASTE_YOUR_KEY
RESEND_FROM_EMAIL=noreply@mentalpath.ca
TWILIO_ACCOUNT_SID=AC_PASTE_YOUR_SID
TWILIO_AUTH_TOKEN=PASTE_YOUR_TOKEN
TWILIO_PHONE_NUMBER=+1PASTE_NUMBER
NEXT_PUBLIC_APP_URL=http://localhost:3000
NEXT_PUBLIC_APP_NAME=MentalPath
EOF

# ── 5. Set Supabase Edge Function secrets ─────────────────
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "⚠️  After filling in .env.local, run:"
echo ""
echo "npx supabase secrets set --project-ref hkhwgbkijepsxtixdmrs \\"
echo "  ANTHROPIC_API_KEY=\$ANTHROPIC_API_KEY \\"
echo "  STRIPE_WEBHOOK_SECRET=\$STRIPE_WEBHOOK_SECRET \\"
echo "  STRIPE_SOLO_PRICE_ID=\$STRIPE_SOLO_PRICE_ID \\"
echo "  STRIPE_GROUP_PRICE_ID=\$STRIPE_GROUP_PRICE_ID"
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "✅ Bootstrap complete! Next steps:"
echo ""
echo "  1. Get your Supabase keys:"
echo "     → https://supabase.com/dashboard/project/hkhwgbkijepsxtixdmrs/settings/api"
echo ""
echo "  2. Fill in .env.local (open it: code .env.local)"
echo ""
echo "  3. Set up Stripe:"
echo "     → https://dashboard.stripe.com/test/products"
echo "     Create Solo plan: \$49 CAD/month"
echo "     Create Group plan: \$79 CAD/month per seat"
echo ""
echo "  4. Get Anthropic API key:"
echo "     → https://console.anthropic.com/settings/keys"
echo ""
echo "  5. Start local dev:"
echo "     npm run dev"
echo ""
echo "  6. Set Supabase secrets (Edge Functions need them):"
echo "     npx supabase secrets set --project-ref hkhwgbkijepsxtixdmrs ANTHROPIC_API_KEY=..."
echo ""
echo "  7. Test AI note assist endpoint:"
echo "     curl -X POST https://hkhwgbkijepsxtixdmrs.supabase.co/functions/v1/ai-note-assist \\"
echo "       -H 'Authorization: Bearer YOUR_USER_JWT' \\"
echo "       -H 'Content-Type: application/json' \\"
echo "       -d '{\"session_id\":\"00000000-0000-0000-0000-000000000020\",\"note_format\":\"DAP\",\"section_1\":\"test data\",\"section_2\":\"test assessment\",\"section_3\":\"test plan\"}'"
echo ""
echo "🍁 MentalPath is ready. Canadian servers. PHIPA compliant. Let's go."
